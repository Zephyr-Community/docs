‚ö° Quick Start

# Installation
npm install zephyr-ai

# Basic Usage
import { ZephyrAI } from 'zephyr-ai';

const ai = new ZephyrAI({
    apiKey: 'your-api-key-here',
    mode: 'unrestricted',
    features: ['web-crawling', 'code-generation', 'data-analysis']
});

// Make unlimited requests
const response = await ai.process({
    request: 'Your complex request here',
    type: 'any-type'
});
